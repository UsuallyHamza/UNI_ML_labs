\section*{Part B: Logistic Regression and Metrics}

\begin{tcolorbox}
\textbf{Task 3: Train Logistic Regression}

Train a logistic regression model using cross-entropy loss:

\[
L = - \frac{1}{N} \sum_{i=1}^{N} \left[ y_i \log(p_i) + (1-y_i)\log(1-p_i) \right]
\]

\vspace{2cm}

\vspace{2cm}
\end{tcolorbox}

\vspace{6pt}

\begin{tcolorbox}
\textbf{Task 4: Evaluate Accuracy and F1 Score}

Compute:

\[
\text{Accuracy} = \frac{TP + TN}{TP + TN + FP + FN}
\]

\[
\text{Precision} = \frac{TP}{TP + FP}, \quad
\text{Recall} = \frac{TP}{TP + FN}
\]

\[
F1 = \frac{2 \cdot (\text{Precision} \cdot \text{Recall})}{\text{Precision} + \text{Recall}}
\]

Compare results for both datasets.

\vspace{2cm}

\vspace{2cm}
\end{tcolorbox}
